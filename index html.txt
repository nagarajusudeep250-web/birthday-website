    <script>
        // --- INITIALIZATION ---
        let currentStep = 1;
        const totalSteps = 5;

        // --- THREE.JS 3D HEARTS ---
        const scene = new THREE.Scene();
        const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
        const renderer = new THREE.WebGLRenderer({ canvas: document.getElementById('three-canvas'), alpha: true, antialias: true });
        renderer.setSize(window.innerWidth, window.innerHeight);

        const heartsGroup = new THREE.Group();
        scene.add(heartsGroup);

        const heartShape = new THREE.Shape();
        heartShape.moveTo(25, 25);
        heartShape.bezierCurveTo(25, 25, 20, 0, 0, 0);
        heartShape.bezierCurveTo(-30, 0, -30, 35, -30, 35);
        heartShape.bezierCurveTo(-30, 55, -10, 77, 25, 95);
        heartShape.bezierCurveTo(60, 77, 80, 55, 80, 35);
        heartShape.bezierCurveTo(80, 35, 80, 0, 50, 0);
        heartShape.bezierCurveTo(35, 0, 25, 25, 25, 25);

        const extrudeSettings = { depth: 8, bevelEnabled: true, bevelSegments: 2, steps: 2, bevelSize: 1, bevelThickness: 1 };
        const heartGeometry = new THREE.ExtrudeGeometry(heartShape, extrudeSettings);
        heartGeometry.scale(0.015, -0.015, 0.015); // Adjust size and flip

        const hearts = [];
        for(let i = 0; i < 25; i++) {
            const material = new THREE.MeshPhongMaterial({ 
                color: new THREE.Color(`hsl(${Math.random() * 20 + 340}, 80%, 60%)`),
                shininess: 100,
                specular: 0x444444
            });
            const heart = new THREE.Mesh(heartGeometry, material);
            heart.position.set(
                (Math.random() - 0.5) * 15,
                (Math.random() - 0.5) * 10,
                (Math.random() - 0.5) * 10
            );
            heart.rotation.set(Math.random() * Math.PI, Math.random() * Math.PI, 0);
            const scale = Math.random() * 0.5 + 0.5;
            heart.scale.set(scale, scale, scale);
            
            // Custom properties for animation
            heart.userData = {
                speed: Math.random() * 0.01 + 0.005,
                angle: Math.random() * Math.PI * 2
            };
            
            hearts.push(heart);
            heartsGroup.add(heart);
        }

        const light = new THREE.DirectionalLight(0xffffff, 1);
        light.position.set(1, 1, 5);
        scene.add(light);
        scene.add(new THREE.AmbientLight(0xffffff, 0.5));
        camera.position.z = 10;

        function animateThree() {
            requestAnimationFrame(animateThree);
            hearts.forEach(heart => {
                heart.rotation.y += heart.userData.speed;
                heart.rotation.x += heart.userData.speed * 0.5;
                heart.position.y += Math.sin(Date.now() * 0.001 + heart.userData.angle) * 0.002;
            });
            renderer.render(scene, camera);
        }
        animateThree();

        // --- NAVIGATION LOGIC ---
        function nextStep(step) {
            const currentEl = document.getElementById(`step-${currentStep}`);
            const nextEl = document.getElementById(`step-${step}`);

            gsap.to(currentEl, {
                opacity: 0,
                scale: 0.8,
                duration: 0.6,
                onComplete: () => {
                    currentEl.classList.remove('active');
                    nextEl.classList.add('active');
                    
                    gsap.fromTo(nextEl, 
                        { opacity: 0, scale: 1.2 },
                        { opacity: 1, scale: 1, duration: 0.8, ease: "back.out(1.7)" }
                    );

                    // Update Progress
                    currentStep = step;
                    const progress = ((step - 1) / (totalSteps - 1)) * 100;
                    gsap.to("#progress-bar", { width: `${progress}%`, duration: 0.5 });
                }
            });
        }

        // --- POLAROID INTERACTION ---
        const polaroid = document.getElementById('memory-polaroid');
        document.addEventListener('mousemove', (e) => {
            if(currentStep !== 4) return;
            const { clientX, clientY } = e;
            const centerX = window.innerWidth / 2;
            const centerY = window.innerHeight / 2;
            const rotateX = (clientY - centerY) / 20;
            const rotateY = (centerX - clientX) / 20;
            
            gsap.to(polaroid, {
                rotationY: rotateY,
                rotationX: rotateX,
                duration: 0.5,
                ease: "power2.out"
            });
        });

        // --- GRAND FINALE ---
        function startFinale() {
            const btn = document.getElementById('celebrate-btn');
            const msg = document.getElementById('final-message');
            
            // 1. Disable & Fade Button
            btn.style.pointerEvents = 'none';
            gsap.to(btn, { opacity: 0, y: 20, duration: 0.5 });

            // 2. Show Final Text
            setTimeout(() => {
                msg.style.display = 'block';
                gsap.fromTo(msg, { opacity: 0, y: 20 }, { opacity: 1, y: 0, duration: 1 });
            }, 600);

            // 3. Confetti Explosion
            const duration = 5 * 1000;
            const animationEnd = Date.now() + duration;
            const defaults = { startVelocity: 30, spread: 360, ticks: 60, zIndex: 0 };

            function randomInRange(min, max) {
              return Math.random() * (max - min) + min;
            }

            const interval = setInterval(function() {
              const timeLeft = animationEnd - Date.now();

              if (timeLeft <= 0) {
                return clearInterval(interval);
              }

              const particleCount = 50 * (timeLeft / duration);
              
              // Side bursts
              confetti({ ...defaults, particleCount, origin: { x: randomInRange(0.1, 0.3), y: Math.random() - 0.2 } });
              confetti({ ...defaults, particleCount, origin: { x: randomInRange(0.7, 0.9), y: Math.random() - 0.2 } });
              
              // Emoji confetti
              confetti({
                particleCount: 2,
                spread: 60,
                origin: { y: 0.6 },
                shapes: ['circle'],
                colors: ['#ff4d94', '#ff1a1a', '#ffffff'],
                scalar: 2
              });
            }, 250);

            // 4. Three.js Hearts Grand Exit
            hearts.forEach((heart, i) => {
                gsap.to(heart.position, {
                    x: (Math.random() - 0.5) * 50,
                    y: 20, // Float up off screen
                    z: (Math.random() - 0.5) * 50,
                    duration: 4,
                    delay: i * 0.05,
                    ease: "power2.in"
                });
                gsap.to(heart.material, {
                    opacity: 0,
                    transparent: true,
                    duration: 3,
                    delay: i * 0.05 + 1
                });
            });
        }

        // Pulse effect for the heart icon
        gsap.to(".beat", {
            scale: 1.2,
            duration: 0.8,
            repeat: -1,
            yoyo: true,
            ease: "power1.inOut"
        });

        // Resize handler
        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });
    </script>
